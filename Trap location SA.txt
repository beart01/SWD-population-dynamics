def accumulate_dd(min_temp, max_temp, lower_limit, upper_limit): # This functions determines
# the degree days generated in each day, based on a sine function that uses max and min
# temps to simulate temperature fluctuations throughout the day. If the temperature at a given
# hour is above the upper_limit, the upper_limit is used to accumulate degree hours. If the temperature
# at any given hour is belor the lower_limit, no ddegree hours are accumulated. The function sums the degree
# hours throughout the day to determine the degree days acumulated for any given day, and returns that value
# to be used when this function is called by other operations.
    daily_amplitude = (max_temp - min_temp)/2
    daily_avg = (max_temp + min_temp)/2
    hourly_temp = []
    hourly_dd = []
    daily_dd = 0
    for j in range(24):
        temp_t = daily_avg + (daily_amplitude * math.sin(2 *  math.pi * (j - 12)/ 24))
        hourly_temp.append(temp_t)
        if lower_limit <= temp_t and temp_t <= upper_limit:
            deg_hour = (temp_t - lower_limit)/24
            hourly_dd.append(deg_hour)
            daily_dd = daily_dd + deg_hour
        elif temp_t > upper_limit:
            deg_hour = upper_limit / 24
            hourly_dd.append(deg_hour)
            daily_dd = daily_dd + deg_hour
        else:
            hourly_dd.append(0)
    return(daily_dd)

def surv(dd): #this function returns probability survival accorting to the
    # value of the index in 'vector_dds'.
    surv = np.exp((-rate / shape) * (np.exp(shape * dd) - 1))
    return(surv)

def fecundity(dd): # this function returns fecundity
    # of a female at physiological age based on the value at the index of 'vector_dds'; approximates Fig. 2 in Wiman et al. (2016).
    eggs = fecundity_scalar * cauchy_constant / (cauchy_scale * math.pi * (1 + ((dd - cauchy_loc) / cauchy_scale) ** 2))
    return(eggs)

def temp_fecundity(temp, center = temp_fecundity_center, height = 1, sigma = 5): # this function returns proportion
    # theoretical fecundity as a function of temperature
    variance = sigma ** (2)
    temp_fecundity_prop = height * math.exp(-((temp - center) ** 2)/(2 * variance))
    return(temp_fecundity_prop)

def all_hosts(i, height = crop_amp, adjust = adj, amplitude = amp): # This function calculates total host
# availability, including rabbiteye and shouthern highbush crop hosts, and non-crop hosts. The peak hosts values in each
# season are used to scale total hosts so that highest value each year is 1.
    x = day_of_year[i]
    re_sigma = re_duration / 4
    shb_sigma = shb_duration / 4
    non_crop_normal = adjust + amplitude * math.sin(2 *  math.pi * (x - non_crop_center)/ 365) # non-crop mid and late season hosts dist
    early_variance = early_sigma ** (2) # variance for early non-crop hosts
    early_noncrop_hosts = (early_amplitude* math.exp(-((x - early_center) ** 2)/(2 * early_variance))) # early non-crop hosts
    all_noncrop = non_crop_adjust * (non_crop_normal + early_noncrop_hosts) # total noncrop hosts for day x
    re = height * math.exp(-((x - rabbiteye_centers[i]) ** 2) / (2 * (re_sigma ** 2)))
    shb = height * math.exp(-((x - s_highbush_centers[i]) ** 2) / (2 * (shb_sigma ** 2)))
    crop_hosts = re + shb
    total_hosts = (all_noncrop + crop_hosts)
    return(total_hosts)

def daily_vector():
    for i in range(len(max_temps)):# This loop generates eggs, produces probability 
    # of survival, and removal due to dd_death, for each day. All list/array indexes
    # are syncronized and functions are tied to the values at each index 
    # in the array 'vector_dds', which indicates the accumulated degree days for each cohort.
    # Each cohort is represented by it's location (index) in various arrays.
    # Taken together, the information at any given index in all of the arrays/vectors will describe a 
    # cohort, specifying information on initial cohort size, current size, and accumulated degree days,
    # in 'initial_cohort_size', 'pop_vector', and 'vector_dds', respectively. This function iterates
    # through the days for which data are listed in the temperature arrays at the begining of the code.
        global vector_dds, pop_vector, initial_cohort_size, total_pop, log_total, pop_proportions, neweggs
        index = []
        surv_vector = []
        eggs = 0 # a counter that sets the eggs produced each day to zero
        if host_reference == 'actual':
            host_avail = unadjusted_hosts[i] # provides a host availability adjustment
        if host_reference == 'relative':
            host_avail = relative_hosts[i]
        tf = temp_fecundity(avg_temps[i])
        immatures = 0
        host_swd_interaction = (attack_rate * host_avail) / (1 + attack_rate * host_avail)
        for j in range(len(vector_dds)):
            if vector_dds[j] < pupae_dds:
                immatures += pop_vector[j]
            else:
                continue
        density_fecundity = 1 - (immatures / (carrying_capacity * host_avail))
        for j in range(len(vector_dds)):
            if vector_dds[j] >= repro_dds and repro_low <= avg_temps[i] <= repro_high: # this loop 
            # generates eggs for each cohort as long as the conditions are met: accumulated dds for 
            # that cohort is at least 210 AND the daily average falls within the range of 
            # the reproductive thresholds. Eggs for a cohort are produced according to the fecundity function,
            # driven by dds of that cohort.            
                max_eggs = pop_vector[j] * fecundity(vector_dds[j]) # Theoretical maximum eggs produced by a cohort
                fecundity_mods = tf * host_swd_interaction * density_fecundity # Modifications to fecundity
                new_eggs = 0.5 * fecundity_mods * max_eggs # assumes sex ratio is 50% female; actual eggs produced by each cohort (realized fecundity)
                eggs += np.round(new_eggs) # counter that adds the number of eggs from each cohort
            else:
                continue
        host_change.append(host_avail - unadjusted_hosts[i - 1]) # This line of code tracks the direction of change
        # for host availability
        # This line of code updates the population and degree day arrays/vectors for a new cohort.
        pop_vector = np.concatenate([[eggs], pop_vector])
        initial_cohort_size = np.concatenate([[eggs], initial_cohort_size])
        vector_dds = np.concatenate([[0], vector_dds])
        neweggs.append(eggs)
        
        ''' The folowing four lines of code are counters for the life stages in
            the population at each day. ''' 
        eggses = 0
        larvae = 0 
        pupae = 0 
        adults = 0 
        for m in range(len(vector_dds)): # This loop partitions the cohorts in 'pop_vector' into
        # life stages according to the accumulated dds for each cohort, corresponding to the synchronized 
        # indices with 'vector_dds'.
            if vector_dds[m] <= egg_dds:
                eggses += pop_vector[m]
            if egg_dds < vector_dds[m] <= larvae_dds:
                larvae += pop_vector[m]
            if larvae_dds < vector_dds[m] <= pupae_dds:
                pupae += pop_vector[m]
            if pupae_dds < vector_dds[m]:
                adults += pop_vector[m]
        total = eggses + larvae + pupae + adults
        ln_total = np.log(total + 1)
        total_pop.append(total)
        log_total.append(ln_total)
        structured_pop[0].append(eggses)
        structured_pop[1].append(larvae)
        structured_pop[2].append(pupae)
        structured_pop[3].append(adults)
        
        vector_dds = vector_dds + accumulate_dd(min_temps[i], max_temps[i], t_low, t_high) # this line
        # adds the degree days accumulated for this day to all the cohort dds in the array 'vector_dds'.
        for k in range(len(vector_dds)): # This loop calculates probability of survival to the accumulated
        # dds for each cohort.
            surv_vector.append(surv(vector_dds[k]))    
        pop_vector = np.round(np.multiply(initial_cohort_size, surv_vector)) # This line reduces survival
        # for each cohort as determined by the total dds accumulated for that cohort.
        for l in range(len(vector_dds)): # This loop identifies any cohorts that have
        # reached or exceeded degree-day death (dd_death).
            if vector_dds[l] > dd_death:
                index.append([l])
        vector_dds = np.delete(vector_dds, index)
        pop_vector = np.delete(pop_vector, index)
        initial_cohort_size = np.delete(initial_cohort_size, index)

        prop_eggs = eggses / total
        prop_larvae = larvae / total
        prop_pupae = pupae / total
        prop_adults = adults / total
        proportions = [prop_eggs, prop_larvae, prop_pupae, prop_adults]
        pop_proportions.append(proportions)
        eggs2adults.append(eggs / adults)
        eggses2adults.append(eggses / adults)
    
daily_vector() # This line of code executes the function 'daily_vector' to run the simulation.
